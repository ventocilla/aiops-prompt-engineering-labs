## Generated Dockerfile

```dockerfile
# syntax=docker/dockerfile:1.7

FROM python:3.12-slim-bookworm AS runtime

ENV PYTHONDONTWRITEBYTECODE=1 \
    PYTHONUNBUFFERED=1 \
    PIP_NO_CACHE_DIR=1 \
    PORT=8080

WORKDIR /app

# Cria usuário non-root
RUN groupadd --system appgroup \
    && useradd --system --gid appgroup --home-dir /app appuser

# Copia dependências primeiro para otimizar cache
COPY requirements.txt .

RUN pip install --upgrade pip \
    && pip install --no-cache-dir -r requirements.txt

# Copia apenas o código necessário da aplicação
COPY app.py .
COPY lib/ ./lib/

# Ajusta permissões
RUN chown -R appuser:appgroup /app

USER appuser

EXPOSE 8080

# Healthcheck simples para Kubernetes/Docker.
# Idealmente, a aplicação deve expor um endpoint /health.
HEALTHCHECK --interval=30s --timeout=5s --start-period=10s --retries=3 \
    CMD python -c "import urllib.request; urllib.request.urlopen('http://127.0.0.1:8080/health')" || exit 1

CMD ["gunicorn", "--bind", "0.0.0.0:8080", "--workers", "4", "app:app"]

```
